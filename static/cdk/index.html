Stack Definition (lib/static-website-stack.ts)

import * as cdk from 'aws-cdk-lib';
import { Construct } from 'constructs';
import * as s3 from 'aws-cdk-lib/aws-s3';
import * as cloudfront from 'aws-cdk-lib/aws-cloudfront';
import * as origins from 'aws-cdk-lib/aws-cloudfront-origins';
import * as route53 from 'aws-cdk-lib/aws-route53';
import * as targets from 'aws-cdk-lib/aws-route53-targets';
import * as acm from 'aws-cdk-lib/aws-certificatemanager';

export class StaticWebsiteStack extends cdk.Stack {
  constructor(scope: Construct, id: string, props?: cdk.StackProps) {
    super(scope, id, props);

    // Create S3 bucket for website content
    const websiteBucket = new s3.Bucket(this, 'WebsiteBucket', {
      bucketName: 'my-static-website-content',
      blockPublicAccess: s3.BlockPublicAccess.BLOCK_ALL,
      removalPolicy: cdk.RemovalPolicy.DESTROY,
      autoDeleteObjects: true,
      encryption: s3.BucketEncryption.S3_MANAGED,
    });

    // Create CloudFront origin access identity
    const originAccessIdentity = new cloudfront.OriginAccessIdentity(this, 'OAI', {
      comment: 'Access to website bucket',
    });
    
    websiteBucket.grantRead(originAccessIdentity);

    // Create CloudFront distribution
    const distribution = new cloudfront.Distribution(this, 'Distribution', {
      defaultBehavior: {
        origin: new origins.S3Origin(websiteBucket, {
          originAccessIdentity,
        }),
        viewerProtocolPolicy: cloudfront.ViewerProtocolPolicy.REDIRECT_TO_HTTPS,
        cachePolicy: cloudfront.CachePolicy.CACHING_OPTIMIZED,
      },
      defaultRootObject: 'index.html',
      errorResponses: [
        {
          httpStatus: 404,
          responseHttpStatus: 200,
          responsePagePath: '/index.html',
        },
      ],
      priceClass: cloudfront.PriceClass.PRICE_CLASS_100,
    });

    // Output the CloudFront URL
    new cdk.CfnOutput(this, 'DistributionDomainName', {
      value: distribution.distributionDomainName,
      description: 'CloudFront Distribution Domain Name',
    });
  }
}

S3 Bucket Configuration

const websiteBucket = new s3.Bucket(this, 'WebsiteBucket', {
  bucketName: 'my-static-website-content',
  blockPublicAccess: s3.BlockPublicAccess.BLOCK_ALL,
  removalPolicy: cdk.RemovalPolicy.DESTROY,
  autoDeleteObjects: true,
  encryption: s3.BucketEncryption.S3_MANAGED,
  
  // Optional: Enable server access logs
  serverAccessLogsPrefix: 'access-logs/',
  
  // Optional: Define lifecycle rules for cost optimization
  lifecycleRules: [
    {
      id: 'DeleteOldLogs',
      enabled: true,
      prefix: 'access-logs/',
      expiration: cdk.Duration.days(90),
    },
  ],
});

// Grant read access to CloudFront
websiteBucket.grantRead(originAccessIdentity);

CloudFront Distribution

const distribution = new cloudfront.Distribution(this, 'Distribution', {
  defaultBehavior: {
    origin: new origins.S3Origin(websiteBucket, {
      originAccessIdentity,
    }),
    viewerProtocolPolicy: cloudfront.ViewerProtocolPolicy.REDIRECT_TO_HTTPS,
    cachePolicy: cloudfront.CachePolicy.CACHING_OPTIMIZED,
    
    // Optional: Enable compression
    compress: true,
    
    // Optional: Define allowed HTTP methods
    allowedMethods: cloudfront.AllowedMethods.ALLOW_GET_HEAD,
  },
  
  // Set default document
  defaultRootObject: 'index.html',
  
  // For SPA routing - return index.html for 404 errors
  errorResponses: [
    {
      httpStatus: 404,
      responseHttpStatus: 200,
      responsePagePath: '/index.html',
    },
  ],
  
  // Optimize for cost by limiting edge locations
  priceClass: cloudfront.PriceClass.PRICE_CLASS_100,
  
  // Enable logging
  enableLogging: true,
  logBucket: new s3.Bucket(this, 'LogBucket', {
    removalPolicy: cdk.RemovalPolicy.DESTROY,
    autoDeleteObjects: true,
  }),
  logFilePrefix: 'cloudfront-logs/',
});
